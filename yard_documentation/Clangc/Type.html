Return an array that contains all the types for the argument of the function that is related to the current type. If the current type is not a function, it returns an empty array.

def arg_types
  num = num_arg_types
  return [] if num == -1

  types = []
  for i in 0..(num - 1) do
    types << arg_type(i)
  end
  types
end
