c_Module_get_ast_file(VALUE self)

    VALUE
c_Module_get_top_level_header(VALUE self, VALUE translation_unit, VALUE index)
{
    Module_t *m;
    Data_Get_Struct(self, Module_t, m);
    TranslationUnit_t *t;
    CHECK_ARG_TYPE(translation_unit, TranslationUnit);
    Data_Get_Struct(translation_unit, TranslationUnit_t, t);
    unsigned int c_index = NUM2UINT(index);
    VALUE header;
    File_t *f;
    R_GET_CLASS_DATA("Clangc", File, header, f);
    f->data = clang_Module_getTopLevelHeader(t->data, m->data, c_index);
    if (f->data)
        return header;
    else
        return Qnil;
}
    Clangc::Module#top_level_headers(Clangc::TranslationUnit) → Array click to toggle source

    Return an array that contains all the Clangc::File corresponding to the related toplevel headers. If the current cursor is not a module, it returns an empty array.

    # File lib/clangc.rb, line 231
def top_level_headers(tu)
  num = num_top_level_headers(tu)
  headers = []
  return headers if num < 1

  for i in 0..(num - 1) do
    headers << top_level_header(tu, i)
  end
  headers
end
